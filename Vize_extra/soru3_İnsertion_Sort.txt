void insertionSort(int array[SIZE])
{
    int i,j,temp;

    for(i=1;i<SIZE;i++)
    {
        j=i;
        while( j>0 && array[j]<array[j-1] )
        {
            temp=array[j];
            array[j]=array[j-1];
            array[j-1]=temp;
            j--;
        }
    }
}
